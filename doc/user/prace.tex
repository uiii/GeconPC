\newcommand{\newterm}[2]{%
    \define@cmdkeys{term}[term@#1@]{%
        description,plural,sort,see,%
        1pad,2pad,3pad,4pad,5pad,6pad,7pad%
    }[none]
    \setkeys{term}{#2}
    \newglossaryentry{#1}{%
        name={\termdef{#1}{1pad}},%
        description={\termdef{#1}{description}},%
        plural={\termdef{#1}{plural}},%
        sort={\termdef{#1}{sort}},%
        see={\termdef{#1}{see}},%
        first={\itshape\termdef{#1}{1pad}}%
    }
}
